config.py

import json
from pathlib import Path

CONFIG_FILE = Path("config.json")

def load_config():
    if CONFIG_FILE.exists():
        return json.loads(CONFIG_FILE.read_text())
    return {"contracts": [], "telegram_token": "", "telegram_chat_id": ""}
cli.py (main entry)

import typer
from .config import load_config, CONFIG_FILE
import json

app = typer.Typer()

@app.command()
def add(address: str, telegram_token: str = None, chat_id: str = None):
    config = load_config()
    config["contracts"].append(address)
    if telegram_token: config["telegram_token"] = telegram_token
    if chat_id: config["telegram_chat_id"] = chat_id
    CONFIG_FILE.write_text(json.dumps(config, indent=2))
    print(f"Added contract {address}")

@app.command()
def start():
    from .monitor import run_monitor
    run_monitor()

if __name__ == "__main__":
    app()

alerts.py

from telegram import Bot

async def send_alert(message: str):
    config = load_config()
    bot = Bot(token=config["telegram_token"])
    await bot.send_message(chat_id=config["telegram_chat_id"], text=message)
